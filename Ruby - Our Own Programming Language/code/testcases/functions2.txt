//Recursive function calculating factorial numbers
fun fact(n)
{
	if (n <= 1)
	{
		return 1;
	}
	return n * fact(n - 1);  //Here's a problem with negative numbers
}

output(fact(1) . "=1?");
output(fact(2) . "=2?");
output(fact(5) . "=120?");
output(fact(7) . "=5040?");
